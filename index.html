// TikTok Event Code - باشترکراو بۆ ئەوەی دڵنیابیت داتا دەنێرێت function trackContact(platform, event) { // ڕێگری لە کردنەوەی یەکسەری لینکەکە دەکات بۆ چرکەیەک event.preventDefault(); var url = event.currentTarget.href; ttq.track('Contact', { content_name: platform, content_category: 'Sponsor Link' }); // دوای ٢٠٠ میلی چرکە لینکەکە دەکاتەوە تا دڵنیا بین تیکتۆک داتاکەی بردوە setTimeout(function() { window.location.href = url; }, 200); }
